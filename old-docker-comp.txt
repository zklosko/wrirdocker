  webdav:
    image: twizzel/webdav
    restart: always
    ports:
      - target: 443
        published: 443
        protocol: tcp
        mode: host
    environment:
      AUTH_TYPE: Basic
      SSL_CERT: selfsigned
      SERVER_NAMES: files.wrir.org
      REALM: WRIR Network Drives
    volumes:
      - ./webdav/mounts:/var/lib/dav/data
      - ./webdav/user.passwd:/user.passwd
